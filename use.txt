@SpringBootApplication
@ComponentScan(
    basePackages = "com.example.dependency.package",
    includeFilters = @ComponentScan.Filter(
        type = FilterType.ASSIGNABLE_TYPE,
        value = {
            com.example.dependency.package.ClassA.class,
            com.example.dependency.package.ClassB.class
        }
    ),
    useDefaultFilters = false // Disable default scanning
)
public class Application {
    public static void main(String[] args) {
        SpringApplication.run(Application.class, args);
    }
}
